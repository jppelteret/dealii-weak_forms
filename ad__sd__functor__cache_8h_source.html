60  template <int dim, int spacedim>
61  static void
62  initialize(MeshWorker::ScratchData<dim, spacedim> &scratch_data,
63  const AD_SD_Functor_Cache *const ad_sd_functor_cache)
64  {
65  // If the user has provided a persistent data then we can leverage that
66  // for the cache. Otherwise, the general data storage provided by the
67  // scratch data object can simply reference itself as the cache.
68  if (ad_sd_functor_cache == nullptr)
69  return;
70 
71  GeneralDataStorage &scratch_cache =
72  scratch_data.get_general_data_storage();
73 
74  // Register the user cache in a central and always accessible place.
75  AD_SD_Functor_Cache &user_cache =
76  const_cast<AD_SD_Functor_Cache &>(*ad_sd_functor_cache);
77  scratch_cache.add_unique_reference<AD_SD_Functor_Cache>(
78  get_name_ad_sd_cache(), user_cache);
79  }

81  template <int dim, int spacedim>
82  static void
83  bind_user_cache_to_thread(
84  MeshWorker::ScratchData<dim, spacedim> &scratch_data)
85  {
86  if (has_user_cache(scratch_data) == false)
87  return;
88 
89  // We must ensure that we do not try to evaluate from multiple threads
90  // at once.
91  AD_SD_Functor_Cache &user_cache = get_user_cache(scratch_data);
92 
93  // We use an infinite loop because we cannot be sure when the next
94  // entry will become free for use.
95  while (true)
96  {
97  for (auto &lock_and_cache : user_cache.source_lock_and_cache)
98  {
99  if (lock_and_cache.first.try_lock())
100  {
101  // Now that we've marked this entry as no longer being
102  // available for use, and point the current thread towards
103  // its associated cache data.
104  GeneralDataStorage &data_storage =
105  get_data_storage(scratch_data);
106 
107 #ifdef DEBUG
108  if (data_storage.stores_object_with_name(
109  get_name_active_data_storage()))
110  {
111  Assert(
112  data_storage.get_object_with_name<GeneralDataStorage *>(
113  get_name_active_data_storage()) == nullptr,
114  ExcMessage(
115  "Expected to find an uninitialised pointer to an active data storage object."));
116  }
117 #endif
118 
119  data_storage.add_or_overwrite_copy<GeneralDataStorage *>(
120  get_name_active_data_storage(), &lock_and_cache.second);
121  return;
122  }
123  }
124  }
125  }

127  template <int dim, int spacedim>
128  static void
129  unbind_user_cache_from_thread(
130  MeshWorker::ScratchData<dim, spacedim> &scratch_data,
131  const GeneralDataStorage & cache)
132  {
133  if (has_user_cache(scratch_data) == false)
134  return;
135 
136  AD_SD_Functor_Cache &user_cache = get_user_cache(scratch_data);
137  for (auto &lock_and_cache : user_cache.source_lock_and_cache)
138  {
139  if (&cache == &lock_and_cache.second)
140  {
141  Assert(lock_and_cache.first.try_lock() == false,
142  ExcMessage("Cache entry was not locked upon return."));
143 
144  GeneralDataStorage &data_storage = get_data_storage(scratch_data);
145  Assert(
146  data_storage.stores_object_with_name(
147  get_name_active_data_storage()),
148  ExcMessage(
149  "Expected to find a pointer to an active data storage object."));
150 
151 
152  // Invalidate pointer in common storage and mark this entry as
153  // being available for re-use.
154  data_storage.add_or_overwrite_copy<GeneralDataStorage *>(
155  get_name_active_data_storage(), nullptr);
156  lock_and_cache.first.unlock();
157  return;
158  }
159  }
160 
161  AssertThrow(false,
162  ExcMessage("Source cache not found in this data structure."));
163  }

165  template <int dim, int spacedim>
166  static GeneralDataStorage &
167  get_cache(MeshWorker::ScratchData<dim, spacedim> &scratch_data)
168  {
169  if (has_user_cache(scratch_data))
170  {
171  GeneralDataStorage &data_storage = get_data_storage(scratch_data);
172 
173  Assert(
174  data_storage.stores_object_with_name(
175  get_name_active_data_storage()),
176  ExcMessage(
177  "Expected to find a pointer to an active data storage object."));
178  GeneralDataStorage *active_data_storage =
179  data_storage.get_object_with_name<GeneralDataStorage *>(
180  get_name_active_data_storage());
181 
182  Assert(
183  active_data_storage != nullptr,
184  ExcMessage(
185  "Expected to find an initialised pointer to an active data storage object."));
186  return *active_data_storage;
187  }
188  else
189  {
190  return get_data_storage(scratch_data);
191  }
192  }
