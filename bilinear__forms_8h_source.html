310  template <typename TestSpaceOp_,
311  typename Functor_,
312  typename TrialSpaceOp_,
313  BilinearFormComponentFilter ComponentFilterFlags =
314  BilinearFormComponentFilter::form_components_default>
315  class BilinearForm
316  {
317  static_assert(is_or_has_test_function_op<TestSpaceOp_>::value,
318  "Expected a test function.");
319  static_assert(
320  is_valid_form_functor<Functor_>::value,
321  "Expected the functor to a linear form to either be a SymbolicOp, or that the unary or binary operation does not include a test function or trial solution as a lead operation.");
322  static_assert(!is_symbolic_integral_op<Functor_>::value,
323  "Functor cannot be an integral.");
324  static_assert(is_or_has_trial_solution_op<TrialSpaceOp_>::value,
325  "Expected a trial solution.");
326 
327  public:
328  using TestSpaceOp = TestSpaceOp_;
329  using Functor = Functor_;
330  using TrialSpaceOp = TrialSpaceOp_;

351  std::string
352  as_latex(const SymbolicDecorations &decorator) const
353  {
354  const std::string lbrace = Utilities::LaTeX::l_square_brace();
355  const std::string rbrace = Utilities::LaTeX::r_square_brace();
356 
357  // If the functor is scalar valued, then we need to be a bit careful about
358  // what the test and trial space ops are (e.g. rank > 0)
359  if (Functor::rank == 0)
360  {
361  constexpr unsigned int n_contracting_indices_tt =
362  WeakForms::Utilities::FullIndexContraction<
363  TestSpaceOp,
364  TrialSpaceOp>::n_contracting_indices;
365 
366  const std::string symb_mult_tt =
367  Utilities::LaTeX::get_symbol_multiply(n_contracting_indices_tt);
368  const std::string symb_mult_sclr =
369  Utilities::LaTeX::get_symbol_multiply(Functor::rank);
370 
371  // TODO: has_kronecker_delta_property()
372  // TODO: is_symmetric()
373  return lbrace + test_space_op.as_latex(decorator) + symb_mult_tt +
374  lbrace + functor_op.as_latex(decorator) + symb_mult_sclr +
375  trial_space_op.as_latex(decorator) + rbrace + rbrace;
376  }
377  else
378  {
379  constexpr unsigned int n_contracting_indices_tf =
380  WeakForms::Utilities::FullIndexContraction<TestSpaceOp, Functor>::
381  n_contracting_indices;
382  constexpr unsigned int n_contracting_indices_ft =
383  WeakForms::Utilities::FullIndexContraction<Functor, TrialSpaceOp>::
384  n_contracting_indices;
385  const std::string symb_mult_tf =
386  Utilities::LaTeX::get_symbol_multiply(n_contracting_indices_tf);
387  const std::string symb_mult_ft =
388  Utilities::LaTeX::get_symbol_multiply(n_contracting_indices_ft);
389 
390  // TODO: has_kronecker_delta_property()
391  // TODO: is_symmetric()
392  return lbrace + test_space_op.as_latex(decorator) + symb_mult_tf +
393  functor_op.as_latex(decorator) + symb_mult_ft +
394  trial_space_op.as_latex(decorator) + rbrace;
395  }
