361  template <typename TestSpaceOp_,
362  typename Functor_,
363  typename TrialSpaceOp_,
364  BilinearFormComponentFilter ComponentFilterFlags =
365  BilinearFormComponentFilter::form_components_default>
366  class BilinearForm
367  {
368  static_assert(is_or_has_test_function_op<TestSpaceOp_>::value,
369  "Expected a test function.");
370  static_assert(
371  is_valid_form_functor<Functor_>::value,
372  "Expected the functor to a linear form to either be a SymbolicOp, or that the unary or binary operation does not include a test function or trial solution as a lead operation.");
373  static_assert(!is_symbolic_integral_op<Functor_>::value,
374  "Functor cannot be an integral.");
375  static_assert(is_or_has_trial_solution_op<TrialSpaceOp_>::value,
376  "Expected a trial solution.");
377 
378  public:
379  using TestSpaceOp = TestSpaceOp_;
380  using Functor = Functor_;
381  using TrialSpaceOp = TrialSpaceOp_;
382 
383  explicit BilinearForm(const TestSpaceOp & test_space_op,
384  const Functor & functor_op,
385  const TrialSpaceOp &trial_space_op)
386  : test_space_op(test_space_op)
387  , functor_op(functor_op)
388  , trial_space_op(trial_space_op)
389  , local_contribution_symmetry_flag(false)
390  {}

392  std::string
393  as_ascii(const SymbolicDecorations &decorator) const
394  {
395  const std::string sym = (is_symmetric() ? "sym" : "");
396  const std::string k_delta =
397  (has_kronecker_delta_property(ComponentFilterFlags) ? "D^(IJ)" : "");
398 
399  const auto contract_with_basis =
400  [&decorator](const std::string &component,
401  const bool shape_function_component = false)
402  {
403  const auto &naming = decorator.get_naming_ascii().geometry;
404 
405  // TODO: get this from a VectorFunctor?
406  if (shape_function_component)
407  return " . <" + naming.basis + ">^(" + component + ")";
408  else
409  return " . <" + naming.basis + ">_(" + component + ")";
410  };
411 
412  // From the assembly point of view, the order of the test/trial
413  // function manipulations is important!
414  // See extract_test_shape_component() in assembler_base.h
415  std::string test_op =
416  decorator.brace_term_when_required_ascii(test_space_op);
417  if (ComponentFilterFlags & BilinearFormComponentFilter::multiplicity_I)
418  test_op += contract_with_basis("I", true);
419  if (ComponentFilterFlags & BilinearFormComponentFilter::dof_I_component_i)
420  test_op += contract_with_basis("i");
421  if (ComponentFilterFlags & BilinearFormComponentFilter::dof_I_component_j)
422  test_op += contract_with_basis("j");
423 
424  std::string trial_op =
425  decorator.brace_term_when_required_ascii(trial_space_op);
426  if (ComponentFilterFlags & BilinearFormComponentFilter::multiplicity_J)
427  trial_op += contract_with_basis("J", true);
428  if (ComponentFilterFlags & BilinearFormComponentFilter::dof_J_component_i)
429  trial_op += contract_with_basis("i");
430  if (ComponentFilterFlags & BilinearFormComponentFilter::dof_J_component_j)
431  trial_op += contract_with_basis("j");
432 
433  return sym + "(" + test_op + ", " + functor_op.as_ascii(decorator) +
434  ", " + trial_op + ")" + k_delta;
435  }

437  std::string
438  as_latex(const SymbolicDecorations &decorator) const
439  {
440  const std::string lbrace = Utilities::LaTeX::l_square_brace();
441  const std::string rbrace = Utilities::LaTeX::r_square_brace();
442 
443  const auto contract_with_basis =
444  [&decorator](const std::string &component,
445  unsigned int & contraction_counter,
446  const bool shape_function_component = false)
447  {
448  const std::string symb_mult_tf =
449  Utilities::LaTeX::get_symbol_multiply(1);
450  ++contraction_counter;
451 
452  const auto &naming = decorator.get_naming_latex().geometry;
453 
454  // TODO: get this from a VectorFunctor?
455  if (shape_function_component)
456  return symb_mult_tf + naming.basis + "^{" + component + "}";
457  else
458  return symb_mult_tf + naming.basis + "_{" + component + "}";
459  };
460 
461  // From the assembly point of view, the order of the test/trial
462  // function manipulations is important!
463  // See extract_test_shape_component() in assembler_base.h
464  std::string test_op =
465  decorator.brace_term_when_required_latex(test_space_op);
466  unsigned int n_reduced_indices_tf = 0;
467  if (ComponentFilterFlags & BilinearFormComponentFilter::multiplicity_I)
468  test_op += contract_with_basis("I", n_reduced_indices_tf, true);
469  if (ComponentFilterFlags & BilinearFormComponentFilter::dof_I_component_i)
470  test_op += contract_with_basis("i", n_reduced_indices_tf);
471  if (ComponentFilterFlags & BilinearFormComponentFilter::dof_I_component_j)
472  test_op += contract_with_basis("j", n_reduced_indices_tf);
473  if (n_reduced_indices_tf > 0)
474  test_op = lbrace + test_op + rbrace;
475 
476  std::string trial_op =
477  decorator.brace_term_when_required_latex(trial_space_op);
478  unsigned int n_reduced_indices_ts = 0;
479  if (ComponentFilterFlags & BilinearFormComponentFilter::multiplicity_J)
480  trial_op += contract_with_basis("J", n_reduced_indices_ts, true);
481  if (ComponentFilterFlags & BilinearFormComponentFilter::dof_J_component_i)
482  trial_op += contract_with_basis("i", n_reduced_indices_ts);
483  if (ComponentFilterFlags & BilinearFormComponentFilter::dof_J_component_j)
484  trial_op += contract_with_basis("j", n_reduced_indices_ts);
485  if (n_reduced_indices_ts > 0)
486  trial_op = lbrace + trial_op + rbrace;
487 
488  const bool use_bilinear_form_notation =
489  (decorator.get_formatting_latex().get_integral_format() ==
490  FormattingLaTeX::IntegralFormat::bilinear_form_notation);
491 
492  // If the functor is scalar valued, then we need to be a bit careful about
493  // what the test and trial space ops are (e.g. rank > 0)
494  std::string standard_bilinear_form;
495  if (Functor::rank == 0)
496  {
497  const int n_contracting_indices_tt =
498  WeakForms::Utilities::FullIndexContraction<TestSpaceOp,
499  TrialSpaceOp>::
500  get_reduced_n_contracting_indices(n_reduced_indices_tf,
501  n_reduced_indices_ts);
502  Assert(n_contracting_indices_tt >= 0,
503  ExcMessage("Invalid index contraction"));
504 
505  // Incorporate the extra indices already contracted out as a part of
506  // the bilinear form component filter operations
507  const std::string symb_mult_tt =
508  (use_bilinear_form_notation ?
509  ", " :
510  Utilities::LaTeX::get_symbol_multiply(n_contracting_indices_tt));
511  const std::string symb_mult_sclr =
512  Utilities::LaTeX::get_symbol_multiply(Functor::rank);
513 
514  standard_bilinear_form =
515  test_op + symb_mult_tt +
516  decorator.brace_term_when_required_latex(functor_op) +
517  symb_mult_sclr + trial_op;
518  }
519  else
520  {
521  const int n_contracting_indices_tf =
522  WeakForms::Utilities::FullIndexContraction<TestSpaceOp, Functor>::
523  get_reduced_n_contracting_indices(n_reduced_indices_tf, 0);
524  const int n_contracting_indices_ft =
525  WeakForms::Utilities::FullIndexContraction<Functor, TrialSpaceOp>::
526  get_reduced_n_contracting_indices(0, n_reduced_indices_ts);
527 
528  Assert(n_contracting_indices_tf >= 0,
529  ExcMessage("Invalid index contraction"));
530  Assert(n_contracting_indices_ft >= 0,
531  ExcMessage("Invalid index contraction"));
532 
533  // Incorporate the extra indices already contracted out as a part of
534  // the bilinear form component filter operations
535  const std::string symb_mult_tf =
536  (use_bilinear_form_notation ?
537  ", " :
538  Utilities::LaTeX::get_symbol_multiply(n_contracting_indices_tf));
539  const std::string symb_mult_ft =
540  Utilities::LaTeX::get_symbol_multiply(n_contracting_indices_ft);
541 
542  standard_bilinear_form =
543  test_op + symb_mult_tf +
544  decorator.brace_term_when_required_latex(functor_op) +
545  symb_mult_ft + trial_op;
546  }
547 
548  if (is_symmetric() || has_kronecker_delta_property(ComponentFilterFlags))
549  {
550  const std::string sym = (is_symmetric() ? "^{S}" : "");
551  const std::string k_delta =
552  (has_kronecker_delta_property(ComponentFilterFlags) ?
553  "\\delta^{IJ}" :
554  "");
555 
556  return lbrace + standard_bilinear_form + rbrace + sym + k_delta;
557  }
558  else
559  return standard_bilinear_form;
560  }
