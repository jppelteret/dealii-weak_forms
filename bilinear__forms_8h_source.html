40  template <typename TestSpaceOp_, typename Functor_, typename TrialSpaceOp_>
41  class BilinearForm
42  {
43  static_assert(is_or_has_test_function_op<TestSpaceOp_>::value,
44  "Expected a test function.");
45  static_assert(
46  is_valid_form_functor<Functor_>::value,
47  "Expected the functor to a linear form to either be a SymbolicOp, or that the unary or binary operation does not include a test function or trial solution as a lead operation.");
48  static_assert(!is_symbolic_integral_op<Functor_>::value,
49  "Functor cannot be an integral.");
50  static_assert(is_or_has_trial_solution_op<TrialSpaceOp_>::value,
51  "Expected a trial solution.");
52 
53  public:
54  using TestSpaceOp = TestSpaceOp_;
55  using Functor = Functor_;
56  using TrialSpaceOp = TrialSpaceOp_;

58  explicit BilinearForm(const TestSpaceOp & test_space_op,
59  const Functor & functor_op,
60  const TrialSpaceOp &trial_space_op)
61  : test_space_op(test_space_op)
62  , functor_op(functor_op)

78  std::string
79  as_latex(const SymbolicDecorations &decorator) const
80  {
81  const std::string lbrace = Utilities::LaTeX::l_square_brace();
82  const std::string rbrace = Utilities::LaTeX::r_square_brace();
83 
84  // If the functor is scalar valued, then we need to be a bit careful about
85  // what the test and trial space ops are (e.g. rank > 0)
86  if (Functor::rank == 0)
87  {
88  constexpr unsigned int n_contracting_indices_tt =
89  WeakForms::Utilities::FullIndexContraction<
90  TestSpaceOp,
91  TrialSpaceOp>::n_contracting_indices;
92 
93  const std::string symb_mult_tt =
94  Utilities::LaTeX::get_symbol_multiply(n_contracting_indices_tt);
95  const std::string symb_mult_sclr =
96  Utilities::LaTeX::get_symbol_multiply(Functor::rank);
97 
98  // TODO: has_kronecker_delta_property()
99  // TODO: is_symmetric()
100  return lbrace + test_space_op.as_latex(decorator) + symb_mult_tt +
101  lbrace + functor_op.as_latex(decorator) + symb_mult_sclr +
102  trial_space_op.as_latex(decorator) + rbrace + rbrace;
103  }
104  else
105  {
106  constexpr unsigned int n_contracting_indices_tf =
107  WeakForms::Utilities::FullIndexContraction<TestSpaceOp, Functor>::
108  n_contracting_indices;
109  constexpr unsigned int n_contracting_indices_ft =
110  WeakForms::Utilities::FullIndexContraction<Functor, TrialSpaceOp>::
111  n_contracting_indices;
112  const std::string symb_mult_tf =
113  Utilities::LaTeX::get_symbol_multiply(n_contracting_indices_tf);
114  const std::string symb_mult_ft =
115  Utilities::LaTeX::get_symbol_multiply(n_contracting_indices_ft);
116 
117  // TODO: has_kronecker_delta_property()
118  // TODO: is_symmetric()
119  return lbrace + test_space_op.as_latex(decorator) + symb_mult_tf +
120  functor_op.as_latex(decorator) + symb_mult_ft +
121  trial_space_op.as_latex(decorator) + rbrace;
122  }
