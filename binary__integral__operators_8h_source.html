256  // Can add and subtract integral operations
257  template <typename T>
258  struct is_binary_integral_op<
259  T,
260  typename std::enable_if<
261  is_binary_op<T>::value &&
262  (T::op_code == Operators::BinaryOpCodes::add ||
263  T::op_code == Operators::BinaryOpCodes::subtract) &&
264  (is_integral_op<typename T::LhsOpType>::value &&
265  is_integral_op<typename T::RhsOpType>::value)>::type> : std::true_type
266  {};
267 
268  // Can only multiply integral operations by arithmetic types.
269  template <typename T>
270  struct is_binary_integral_op<
271  T,
272  typename std::enable_if<
273  is_binary_op<T>::value &&
274  T::op_code == Operators::BinaryOpCodes::multiply &&
275  (is_integral_op<typename T::LhsOpType>::value ^
276  is_integral_op<typename T::RhsOpType>::value)>::type> : std::true_type
277  {};
