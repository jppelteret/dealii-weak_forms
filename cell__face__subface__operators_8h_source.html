252  template <typename ResultScalarType, std::size_t width>
253  vectorized_return_type<ResultScalarType, width>
254  operator()(const FEValuesBase<dim, spacedim> & fe_face_values,
255  const types::vectorized_qp_range_t &q_point_range) const
256  {
257  vectorized_return_type<ResultScalarType, width> out;
258  Assert(q_point_range.size() <= width,
259  ExcIndexRange(q_point_range.size(), 0, width));
260 
261  DEAL_II_OPENMP_SIMD_PRAGMA
262  for (unsigned int i = 0; i < q_point_range.size(); ++i)
263  if (q_point_range[i] < fe_face_values.n_quadrature_points)
264  numbers::set_vectorized_values(
265  out,
266  i,
267  this->template operator()<ResultScalarType>(
268  fe_face_values)[q_point_range[i]]);
269 
270  return out;
271  }
