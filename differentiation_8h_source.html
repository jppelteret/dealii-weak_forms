44  // A wrapper for functors that helps produce the correct differential
45  // notation in the output. Without this, the @p SymbolicDiffOp would have
46  // full authority of what is being printed, which means that we have no
47  // way of isolating the boldening (when necessary) the functor that's
48  // being differentiated and the fields with which it's being
49  // differentiated with respect to. All of the differential operation
50  // notation, and the functor and fields would carry the notation
51  // of the underlying symbolic operator for a
52  // [Scalar/Tensor/SymmetricTensor]CacheFunctor.
53  //
54  // The template parameter @p FunctorOp is the functor that has been
55  // differentiated, and the @p FieldOps are the field operators that its
56  // being differentiated with respect to.
57  template <typename SymbolicFunctorOp,
58  typename FunctorOp,
59  typename... FieldOps>
60  class SymbolicDiffOp : public SymbolicFunctorOp
61  {
62  public:
63  SymbolicDiffOp(const SymbolicFunctorOp &symbolic_functor_op,
64  const FunctorOp & functor_op,
65  const FieldOps &...field_ops)
66  : SymbolicFunctorOp(symbolic_functor_op)
67  , functor_op(functor_op)
68  , field_ops(field_ops...)
69  {}
70 
71  std::string
72  as_ascii(const SymbolicDecorations &decorator) const
73  {
74  const auto &naming =
75  decorator.get_naming_ascii().differential_operators;
76  return decorator.decorate_with_operator_ascii(
77  naming.value,
78  decorator.symbolic_op_derivative_as_ascii(functor_op, field_ops));
79  }
80 
81  std::string
82  as_latex(const SymbolicDecorations &decorator) const
83  {
84  const auto &naming =
85  decorator.get_naming_latex().differential_operators;
86  return decorator.decorate_with_operator_latex(
87  naming.value,
88  decorator.symbolic_op_derivative_as_latex(functor_op, field_ops));
89  }
90 
91  // Expose base class definitions
92  using SymbolicFunctorOp::get_update_flags;
93  using SymbolicFunctorOp::operator();
94 
95  private:
96  const FunctorOp functor_op;
97  const std::tuple<FieldOps...> field_ops;
98  };

104  // Differentiate a scalar with respect to another scalar
105  template <typename T, typename U>
106  struct DiffOpResult<
107  T,
108  U,
109  typename std::enable_if<WeakForms::are_scalar_types<T, U>::value>::type>
110  {
111  static constexpr int rank = 0;
112  using scalar_type = typename ProductType<T, U>::type;
113  using type = scalar_type;
114 
115  using Op = WeakForms::ScalarCacheFunctor;
116  template <int dim, int spacedim = dim>
117  using function_type =
118  typename Op::template function_type<scalar_type, dim, spacedim>;
119 
120  // Return symbolic op to functor
121  template <int dim,
122  int spacedim,
123  typename FunctorOp,
124  typename... FieldOps>
125  static auto
126  get_functor(const std::string & symbol_ascii,
127  const std::string & symbol_latex,
128  const function_type<dim, spacedim> &function,
129  const FunctorOp & functor_op,
130  const FieldOps &...field_ops)
131  {
132  const auto symbolic_op =
133  get_operand(symbol_ascii, symbol_latex)
134  .template value<scalar_type, dim, spacedim>(
135  function, UpdateFlags::update_default);
136  using SymbolicOp_t = typename std::decay<decltype(symbolic_op)>::type;
137  return SymbolicDiffOp<SymbolicOp_t, FunctorOp, FieldOps...>(
138  symbolic_op, functor_op, field_ops...);
139  }
140 
141  private:
142  static Op
143  get_operand(const std::string &symbol_ascii,
144  const std::string &symbol_latex)
145  {
146  return Op(symbol_ascii, symbol_latex);
147  }
148  };

519  // Specialization for differentiating a scalar or tensor with respect
520  // to a tuple of fields.
521  // This is only intended for use in a very narrow context, so we only
522  // define the result types for the operation.
523  //
524  // This builds up into a tuple that can be represented as:
525  // [ df/dx_1 , df/dx_2 , ... , df/dx_n ]
526  template <typename T, typename... Us>
527  struct DiffOpResult<T, std::tuple<Us...>, void>
528  {
529  // The result type
530  using type = std::tuple<typename DiffOpResult<T, Us>::type...>;
531  };

533  // Specialization for differentiating a tuple of scalars or tensors
534  // with respect to a tuple of fields.
535  // This is only intended for use in a very narrow context, so we only
536  // define the result types for the operation.
537  //
538  // This builds up into a nested tuple with the following
539  // structure/grouping:
540  // [ [ d^2f/dx_1.dx_1 , d^2f/dx_1.dx_2 , ... , d^2f/dx_1.dx_n ] ,
541  // [ d^2f/dx_2.dx_1 , d^2f/dx_2.dx_2 , ... , d^2f/dx_2.dx_n ] ,
542  // [ ... ] ,
543  // [ d^2f/dx_n.dx_1 , d^2f/dx_n.dx_2 , ... , d^2f/dx_n.dx_n ] ]
544  //
545  // So the outer tuple holds the "row elements", and the inner tuple
546  // the "column elements" for each row.
547  template <typename... Ts, typename... Us>
548  struct DiffOpResult<std::tuple<Ts...>, std::tuple<Us...>, void>
549  {
550  // The result type
551  using type =
552  std::tuple<typename DiffOpResult<Ts, std::tuple<Us...>>::type...>;
553  };
