652  template <typename ResultScalarType>
653  return_type<ResultScalarType>
654  operator()(MeshWorker::ScratchData<dim, spacedim> &scratch_data,
655  const std::vector<SolutionExtractionData<dim, spacedim>>
656  &solution_extraction_data) const
657  {
658  // Follow the recipe described in the documentation:
659  // - Initialize helper.
660  // - Register independent variables and set the values for all fields.
661  // - Extract the sensitivities.
662  // - Use sensitivities in AD functor.
663  // - Register the definition of the total stored energy.
664  // - Compute gradient, linearization, etc.
665  // - Later, extract the desired components of the gradient,
666  // linearization etc.
667 
668  // Note: All user functions have the same parameterization, so we can
669  // use the same ADHelper for each of them. This does not restrict the
670  // user to use the same definition for the energy itself at each QP!
671  ad_helper_type &ad_helper = get_mutable_ad_helper(scratch_data);
672  std::vector<Vector<scalar_type>> &Dpsi =
673  get_mutable_gradients(scratch_data, ad_helper);
674  std::vector<FullMatrix<scalar_type>> &D2psi =
675  get_mutable_hessians(scratch_data, ad_helper);
676 
677  const FEValuesBase<dim, spacedim> &fe_values =
678  scratch_data.get_current_fe_values();
679 
680  // In the HP case, we might traverse between cells with a different
681  // number of quadrature points. So we need to resize the output data
682  // accordingly.
683  if (Dpsi.size() != fe_values.n_quadrature_points ||
684  D2psi.size() != fe_values.n_quadrature_points)
685  {
686  Dpsi.resize(fe_values.n_quadrature_points,
687  Vector<scalar_type>(ad_helper.n_dependent_variables()));
688  D2psi.resize(
689  fe_values.n_quadrature_points,
690  FullMatrix<scalar_type>(ad_helper.n_dependent_variables(),
691  ad_helper.n_independent_variables()));
692  }
693 
694  for (const auto q_point : fe_values.quadrature_point_indices())
695  {
696  ad_helper.reset();
697 
698  // Register the independent variables. The actual field solution at
699  // the quadrature point is fetched from the scratch_data cache. It
700  // is paired with its counterpart extractor, which should not have
701  // any indiced overlapping with the extractors for the other fields
702  // in the field_args.
703  OpHelper_t::ad_register_independent_variables(
704  ad_helper,
705  scratch_data,
706  solution_extraction_data,
707  q_point,
708  get_field_args(),
709  get_field_extractors());
710 
711  // Evaluate the functor to compute the total stored energy.
712  // To do this, we extract all sensitivities and pass them directly
713  // in the user-provided function.
714  const energy_type psi =
715  OpHelper_t::ad_call_function(ad_helper,
716  function,
717  scratch_data,
718  solution_extraction_data,
719  q_point,
720  get_field_extractors());
721 
722  // Register the definition of the total stored energy
723  ad_helper.register_dependent_variable(psi);
724 
725  // Store the output function value, its gradient and linearization.
726  ad_helper.compute_gradient(Dpsi[q_point]);
727  ad_helper.compute_hessian(D2psi[q_point]);
728  }
729  }

1025  template <typename ResultScalarType>
1026  return_type<ResultScalarType>
1027  operator()(MeshWorker::ScratchData<dim, spacedim> &scratch_data,
1028  const std::vector<SolutionExtractionData<dim, spacedim>>
1029  &solution_extraction_data) const
1030  {
1031  // Follow the recipe described in the documentation:
1032  // - Define some independent variables.
1033  // - Compute symbolic expressions that are dependent on the independent
1034  // variables.
1035  // - Create a optimizer to evaluate the dependent functions.
1036  // - Register symbols that represent independent variables.
1037  // - Register symbolic expressions that represent dependent functions.
1038  // - Optimize: Determine computationally efficient code path for
1039  // evaluation.
1040  // - Substitute: Pass the optimizer the numeric values that thee
1041  // independent variables to represent.
1042  // - Extract the numeric equivalent of the dependent functions from the
1043  // optimizer.
1044 
1045  // Note: All user functions have the same parameterization, so on the
1046  // face of it we can use the same BatchOptimizer for each of them. In
1047  // theory the user can encode the QPoint into the energy function: this
1048  // current implementation restricts the user to use the same definition
1049  // for the energy itself at each QP.
1050  const auto initialize_optimizer =
1051  [this](sd_helper_type<ResultScalarType> &batch_optimizer)
1052  {
1053  Assert(batch_optimizer.n_independent_variables() == 0,
1054  ExcMessage(
1055  "Expected the batch optimizer to be uninitialized."));
1056  Assert(batch_optimizer.n_dependent_variables() == 0,
1057  ExcMessage(
1058  "Expected the batch optimizer to be uninitialized."));
1059  Assert(batch_optimizer.values_substituted() == false,
1060  ExcMessage(
1061  "Expected the batch optimizer to be uninitialized."));
1062 
1063  // Create and register field variables (the independent variables).
1064  // We deal with the fields before the user data just in case
1065  // the users try to overwrite these field symbols. It shouldn't
1066  // happen, but this way its not possible to do overwrite what's
1067  // already in the map.
1068  Differentiation::SD::types::substitution_map symbol_map =
1069  OpHelper_t::template sd_get_symbol_map<sd_type>(
1070  get_symbolic_fields());
1071  if (user_symbol_registration_map)
1072  {
1073  Differentiation::SD::add_to_symbol_map(
1074  symbol_map,
1075  OpHelper_t::template sd_call_function<sd_type>(
1076  user_symbol_registration_map, get_symbolic_fields()));
1077  }
1078  batch_optimizer.register_symbols(symbol_map);
1079 
1080  // The next typical few steps that precede function registration
1081  // have already been performed in the class constructor:
1082  // - Evaluate the functor to compute the total stored energy.
1083  // - Compute the first derivatives of the energy function.
1084  // - If there's some intermediate substitution to be done (modifying
1085  // the first derivatives), then do it before computing the second
1086  // derivatives.
1087  // (Why the intermediate substitution? If the first derivatives
1088  // represent the partial derivatives, then this substitution may be
1089  // done to ensure that the consistent linearization is given by the
1090  // second derivatives.)
1091  // - Differentiate the first derivatives (perhaps a modified form)
1092  // to get the second derivatives.
1093 
1094  // Register the dependent variables.
1095  OpHelper_t::template sd_register_functions<sd_type, energy_type>(
1096  batch_optimizer, first_derivatives);
1097  OpHelper_t::template sd_register_functions<sd_type, energy_type>(
1098  batch_optimizer, second_derivatives);
1099  };
1100 
1101  // Create and, if necessary, optimize a BatchOptimizer instance.
1102  // If the user has specified a cache, then this is only done once
1103  // per number of ScratchData times over the entire lifetime of the
1104  // cache.
1105  // If the user has not specified a cache, then this is done once
1106  // per number of ScratchData times for each assembly step (i.e.
1107  // with an additional multiplication factor like number of timesteps
1108  // times number of Newton iterations).
1109  sd_helper_type<ResultScalarType> &batch_optimizer =
1110  get_mutable_sd_batch_optimizer<ResultScalarType>(scratch_data);
1111  if (batch_optimizer.optimized() == false)
1112  {
1113  initialize_optimizer(batch_optimizer);
1114 
1115  // Finalize the optimizer.
1116  // If using the LLVM optimiser in a threaded environment, we have to
1117  // stagger initialisation so as not to cause some race condition
1118  // that leads to a segfault.
1119  if (optimization_method ==
1120  Differentiation::SD::OptimizerType::llvm &&
1121  MultithreadInfo::is_running_single_threaded() == false)
1122  {
1123  static std::mutex mutex;
1124  std::lock_guard<std::mutex> lock(mutex);
1125  batch_optimizer.optimize();
1126  }
1127  else
1128  {
1129  batch_optimizer.optimize();
1130  }
1131  }
1132 
1133  // Check that we've actually got a state that we can do some work with.
1134  Assert(batch_optimizer.n_independent_variables() > 0,
1135  ExcMessage("Expected the batch optimizer to be initialized."));
1136  Assert(batch_optimizer.n_dependent_variables() > 0,
1137  ExcMessage("Expected the batch optimizer to be initialized."));
1138 
1139  std::vector<std::vector<ResultScalarType>>
1140  &evaluated_dependent_functions =
1141  get_mutable_evaluated_dependent_functions<ResultScalarType>(
1142  scratch_data, batch_optimizer);
1143 
1144  const FEValuesBase<dim, spacedim> &fe_values =
1145  scratch_data.get_current_fe_values();
1146 
1147  // In the HP case, we might traverse between cells with a different
1148  // number of quadrature points. So we need to resize the output data
1149  // accordingly.
1150  if (evaluated_dependent_functions.size() !=
1151  fe_values.n_quadrature_points)
1152  {
1153  evaluated_dependent_functions.resize(
1154  fe_values.n_quadrature_points,
1155  std::vector<ResultScalarType>(
1156  batch_optimizer.n_dependent_variables()));
1157  }
1158 
1159  for (const auto q_point : fe_values.quadrature_point_indices())
1160  {
1161  // Substitute the field variables and whatever user symbols
1162  // are defined.
1163  // First we do the values from finite element fields,
1164  // followed by the values for user parameters, etc.
1165  Differentiation::SD::types::substitution_map substitution_map =
1166  OpHelper_t::template sd_get_substitution_map<sd_type,
1167  ResultScalarType>(
1168  scratch_data,
1169  solution_extraction_data,
1170  q_point,
1171  get_symbolic_fields(),
1172  get_field_args());
1173  if (user_substitution_map)
1174  {
1175  Differentiation::SD::add_to_substitution_map(
1176  substitution_map,
1177  user_substitution_map(scratch_data,
1178  solution_extraction_data,
1179  q_point));
1180  }
1181 
1182  // Perform the value substitution at this quadrature point
1183  batch_optimizer.substitute(substitution_map);
1184 
1185  // Extract evaluated data to be retrieved later.
1186  evaluated_dependent_functions[q_point] = batch_optimizer.evaluate();
1187  }
1188  }
