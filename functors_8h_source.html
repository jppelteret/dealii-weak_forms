504  \
505  \
508  static const unsigned int space_dimension = spacedim; \
509  \
510  using scalar_type = ScalarType; \
511  \
512  template <typename ResultScalarType> \
513  using return_type = std::vector<value_type<ResultScalarType>>; \
514  \
515  template <typename ResultScalarType, std::size_t width> \
516  using vectorized_value_type = typename numbers::VectorizedValue< \
517  value_type<ResultScalarType>>::template type<width>; \
518  \
519  template <typename ResultScalarType, std::size_t width> \
520  using vectorized_return_type = typename numbers::VectorizedValue< \
521  value_type<ResultScalarType>>::template type<width>; \
522  \
523  static const enum SymbolicOpCodes op_code = SymbolicOpCodes::value; \
524  \
525  explicit SymbolicOp( \
526  const Op & operand, \
527  const function_type<ScalarType> & function, \
528  const interface_function_type<ScalarType> &interface_function, \
529  const UpdateFlags update_flags) \
530  : operand(operand) \
531  , function(function) \
532  , interface_function(interface_function) \
533  , update_flags(update_flags) \
534  {} \

537  : SymbolicOp( \
538  operand, \
539  [value](const FEValuesBase<dim, spacedim> &, const unsigned int) \
540  { return value; }, \

546  { \
547  const auto &naming = decorator.get_naming_ascii().differential_operators; \
548  return decorator.decorate_with_operator_ascii( \

553  { \
554  const auto &naming = decorator.get_naming_latex().differential_operators; \
555  return decorator.decorate_with_operator_latex( \
556  naming.value, operand.as_latex(decorator)); \

574  template <typename ResultScalarType> \
575  return_type<ResultScalarType> operator()( \
576  const FEValuesBase<dim, spacedim> &fe_values) const \
577  { \
578  return_type<ResultScalarType> out; \
579  out.reserve(fe_values.n_quadrature_points); \
580  \

596  out.emplace_back( \
597  this->template operator()<ResultScalarType>(fe_interface_values, \

599  \
600  return out; \
601  } \

603  \

619  numbers::set_vectorized_values( \
