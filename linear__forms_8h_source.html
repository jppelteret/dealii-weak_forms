74  template <typename TestSpaceOp_, typename Functor_>
75  class LinearForm
76  {
77  static_assert(is_or_has_test_function_op<TestSpaceOp_>::value,
78  "Expected a test function.");
79  static_assert(
80  is_valid_form_functor<Functor_>::value,
81  "Expected the functor to a linear form to either be a SymbolicOp, or that the unary or binary operation does not include a test function or trial solution as a lead operation.");
82  static_assert(!is_symbolic_integral_op<Functor_>::value,
83  "Functor cannot be an integral.");
84

87  using Functor = Functor_;

102  std::string
103  as_latex(const SymbolicDecorations &decorator) const
104  {
105  constexpr unsigned int n_contracting_indices_tf = WeakForms::Utilities::
106  FullIndexContraction<TestSpaceOp, Functor>::n_contracting_indices;
107  const std::string symb_mult_tf =
108  Utilities::LaTeX::get_symbol_multiply(n_contracting_indices_tf);
