188 // Declare below the definitions of operator* so that they can be
189 // recursively used.

194  template <typename ScalarType, typename SymbolicIntegralOp, typename>
195  DEAL_II_ALWAYS_INLINE inline auto
196  ScalarTimesSymbolicIntegralOpImpl(const ScalarType & value,
197  const SymbolicIntegralOp &integral)
198  {
199  using IntegralScalarType = typename SymbolicIntegralOp::ScalarType;
200  return integral.get_integral_operation()
201  .template integrate<IntegralScalarType>(value *
202  integral.get_integrand());
203  }

206  template <typename ScalarType, typename UnaryIntegralOp, typename>
207  DEAL_II_ALWAYS_INLINE inline auto
208  ScalarTimesUnaryIntegralOpImpl(const ScalarType & value,
209  const UnaryIntegralOp &integral)
210  {
211  using namespace WeakForms;
212  using namespace WeakForms::Operators;
213 
214  using IntegrandOpType =
215  decltype(value * std::declval<typename UnaryIntegralOp::OpType>());
216  using OpType = WeakForms::Operators::UnaryOp<IntegrandOpType,
217  UnaryIntegralOp::op_code>;
218 
219  return OpType(value * integral.get_operand());
220  }

223  template <typename ScalarType, typename BinaryIntegralOp, typename>
224  DEAL_II_ALWAYS_INLINE inline auto
225  ScalarTimesBinaryIntegralOpImpl(const ScalarType & value,
226  const BinaryIntegralOp &integral)
227  {
228  using namespace WeakForms;
229  using namespace WeakForms::Operators;
230 
231  using IntegrandLhsOpType =
232  decltype(value * std::declval<typename BinaryIntegralOp::LhsOpType>());
233  using IntegrandRhsOpType =
234  decltype(value * std::declval<typename BinaryIntegralOp::RhsOpType>());
235  using OpType = WeakForms::Operators::BinaryOp<IntegrandLhsOpType,
236  IntegrandRhsOpType,
237  BinaryIntegralOp::op_code>;
238 
239  return OpType(value * integral.get_lhs_operand(),
240  value * integral.get_rhs_operand());
241  }
