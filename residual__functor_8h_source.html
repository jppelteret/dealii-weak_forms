893  template <typename ResultScalarType>
894  return_type<ResultScalarType>
895  operator()(MeshWorker::ScratchData<dim, spacedim> &scratch_data,
896  const std::vector<SolutionExtractionData<dim, spacedim>>
897  &solution_extraction_data) const
898  {
899  // Follow the recipe described in the documentation:
900  // - Initialize helper.
901  // - Register independent variables and set the values for all fields.
902  // - Extract the sensitivities.
903  // - Use sensitivities in AD functor.
904  // - Register the definition of the residual field.
905  // - Compute residual field value, Jacobian, etc.
906  // - Later, extract the desired components of the residual field value,
907  // Jacobian etc.
908 
909  // Note: All user functions have the same parameterization, so we can
910  // use the same ADHelper for each of them. This does not restrict the
911  // user to use the same definition for the field itself at each
912  // QP!
913  ad_helper_type &ad_helper = get_mutable_ad_helper(scratch_data);
914  std::vector<Vector<scalar_type>> &values =
915  get_mutable_values(scratch_data, ad_helper);
916  std::vector<FullMatrix<scalar_type>> &Dvalues =
917  get_mutable_jacobians(scratch_data, ad_helper);
918 
919  const FEValuesBase<dim, spacedim> &fe_values =
920  scratch_data.get_current_fe_values();
921 
922  // In the HP case, we might traverse between cells with a different
923  // number of quadrature points. So we need to resize the output data
924  // accordingly.
925  if (values.size() != fe_values.n_quadrature_points ||
926  Dvalues.size() != fe_values.n_quadrature_points)
927  {
928  values.resize(fe_values.n_quadrature_points,
929  Vector<scalar_type>(
930  ad_helper.n_dependent_variables()));
931  Dvalues.resize(
932  fe_values.n_quadrature_points,
933  FullMatrix<scalar_type>(ad_helper.n_dependent_variables(),
934  ad_helper.n_independent_variables()));
935  }
936 
937  for (const auto q_point : fe_values.quadrature_point_indices())
938  {
939  ad_helper.reset();
940 
941  // Register the independent variables. The actual field solution at
942  // the quadrature point is fetched from the scratch_data cache. It
943  // is paired with its counterpart extractor, which should not have
944  // any indiced overlapping with the extractors for the other fields
945  // in the field_args.
946  OpHelper_t::ad_register_independent_variables(
947  ad_helper,
948  scratch_data,
949  solution_extraction_data,
950  q_point,
951  get_field_args(),
952  get_field_extractors());
953 
954  // Evaluate the functor to compute the residual field value.
955  // To do this, we extract all sensitivities and pass them directly
956  // in the user-provided function.
957  const residual_type residual_field_value =

1300  template <typename ResultScalarType>
1301  return_type<ResultScalarType>
1302  operator()(MeshWorker::ScratchData<dim, spacedim> &scratch_data,
1303  const std::vector<SolutionExtractionData<dim, spacedim>>
1304  &solution_extraction_data) const
1305  {
1306  // Follow the recipe described in the documentation:
1307  // - Define some independent variables.
1308  // - Compute symbolic expressions that are dependent on the independent
1309  // variables.
1310  // - Create a optimizer to evaluate the dependent functions.
1311  // - Register symbols that represent independent variables.
1312  // - Register symbolic expressions that represent dependent functions.
1313  // - Optimize: Determine computationally efficient code path for
1314  // evaluation.
1315  // - Substitute: Pass the optimizer the numeric values that thee
1316  // independent variables to represent.
1317  // - Extract the numeric equivalent of the dependent functions from the
1318  // optimizer.
1319 
1320  // Note: All user functions have the same parameterization, so on the
1321  // face of it we can use the same BatchOptimizer for each of them. In
1322  // theory the user can encode the QPoint into the field function: this
1323  // current implementation restricts the user to use the same definition
1324  // for the field itself at each QP.
1325  const auto initialize_optimizer =
1326  [this](sd_helper_type<ResultScalarType> &batch_optimizer)
1327  {
1328  Assert(batch_optimizer.n_independent_variables() == 0,
1329  ExcMessage(
1330  "Expected the batch optimizer to be uninitialized."));
1331  Assert(batch_optimizer.n_dependent_variables() == 0,
1332  ExcMessage(
1333  "Expected the batch optimizer to be uninitialized."));
1334  Assert(batch_optimizer.values_substituted() == false,
1335  ExcMessage(
1336  "Expected the batch optimizer to be uninitialized."));
1337 
1338  // Create and register field variables (the independent variables).
1339  // We deal with the fields before the user data just in case
1340  // the users try to overwrite these field symbols. It shouldn't
1341  // happen, but this way its not possible to do overwrite what's
1342  // already in the map.
1343  Differentiation::SD::types::substitution_map symbol_map =
1344  OpHelper_t::template sd_get_symbol_map<sd_type>(
1345  get_symbolic_fields());
1346  if (user_symbol_registration_map)
1347  {
1348  Differentiation::SD::add_to_symbol_map(
1349  symbol_map,
1350  OpHelper_t::template sd_call_function<sd_type>(
1351  user_symbol_registration_map,
1352  get_symbolic_fields(),
1353  false /*compute_hash*/));
1354  }
1355  batch_optimizer.register_symbols(symbol_map);
1356 
1357  // The next typical few steps that precede function registration
1358  // have already been performed in the class constructor:
1359  // - Evaluate the functor to compute the total stored field.
1360  // - Compute the first derivatives of the field function.
1361  // - If there's some intermediate substitution to be done (modifying
1362  // the first derivatives), then do it before computing the second
1363  // derivatives.
1364  // (Why the intermediate substitution? If the first derivatives
1365  // represent the partial derivatives, then this substitution may be
1366  // done to ensure that the consistent linearization is given by the
1367  // second derivatives.)
1368  // - Differentiate the first derivatives (perhaps a modified form)
1369  // to get the second derivatives.
1370 
1371  // Register the dependent variables.
1372  OpHelper_t::template sd_register_functions<sd_type, residual_type>(
1373  batch_optimizer, residual);
1374  OpHelper_t::template sd_register_functions<sd_type, residual_type>(
1375  batch_optimizer, first_derivatives);
1376  };
1377 
1378  // Create and, if necessary, optimize a BatchOptimizer instance.
1379  // If the user has specified a cache, then this is only done once
1380  // per number of ScratchData times over the entire lifetime of the
1381  // cache.
1382  // If the user has not specified a cache, then this is done once
1383  // per number of ScratchData times for each assembly step (i.e.
1384  // with an additional multiplication factor like number of timesteps
1385  // times number of Newton iterations).
1386  sd_helper_type<ResultScalarType> &batch_optimizer =
1387  get_mutable_sd_batch_optimizer<ResultScalarType>(scratch_data);
1388  if (batch_optimizer.optimized() == false)
1389  {
1390  initialize_optimizer(batch_optimizer);
1391 
1392  // Finalize the optimizer.
1393  // If using the LLVM optimiser in a threaded environment, we have to
1394  // stagger initialisation so as not to cause some race condition
1395  // that leads to a segfault.
1396  if (optimization_method ==
1397  Differentiation::SD::OptimizerType::llvm &&
1398  MultithreadInfo::is_running_single_threaded() == false)
1399  {
1400  static std::mutex mutex;
1401  std::lock_guard<std::mutex> lock(mutex);
1402  batch_optimizer.optimize();
1403  }
1404  else
1405  {
1406  batch_optimizer.optimize();
1407  }
1408  }
1409 
1410  // Check that we've actually got a state that we can do some work with.
1411  Assert(batch_optimizer.n_independent_variables() > 0,
1412  ExcMessage("Expected the batch optimizer to be initialized."));
1413  Assert(batch_optimizer.n_dependent_variables() > 0,
1414  ExcMessage("Expected the batch optimizer to be initialized."));
1415 
1416  std::vector<std::vector<ResultScalarType>>
1417  &evaluated_dependent_functions =
1418  get_mutable_evaluated_dependent_functions<ResultScalarType>(
1419  scratch_data, batch_optimizer);
1420 
1421  const FEValuesBase<dim, spacedim> &fe_values =
1422  scratch_data.get_current_fe_values();
1423 
1424  // In the HP case, we might traverse between cells with a different
1425  // number of quadrature points. So we need to resize the output data
1426  // accordingly.
1427  if (evaluated_dependent_functions.size() !=
1428  fe_values.n_quadrature_points)
1429  {
1430  evaluated_dependent_functions.resize(
1431  fe_values.n_quadrature_points,
1432  std::vector<ResultScalarType>(
1433  batch_optimizer.n_dependent_variables()));
1434  }
1435 
1436  for (const auto q_point : fe_values.quadrature_point_indices())
1437  {
1438  // Substitute the field variables and whatever user symbols
1439  // are defined.
1440  // First we do the values from finite element fields,
1441  // followed by the values for user parameters, etc.
1442  Differentiation::SD::types::substitution_map substitution_map =
1443  OpHelper_t::template sd_get_substitution_map<sd_type,
1444  ResultScalarType>(
1445  scratch_data,
1446  solution_extraction_data,
1447  q_point,
1448  get_symbolic_fields(),
1449  get_field_args());
1450  if (user_substitution_map)
1451  {
1452  Differentiation::SD::add_to_substitution_map(
1453  substitution_map,
1454  user_substitution_map(scratch_data,
1455  solution_extraction_data,
1456  q_point));
1457  }

1460  batch_optimizer.substitute(substitution_map);
1461

1463  evaluated_dependent_functions[q_point] = batch_optimizer.evaluate();
