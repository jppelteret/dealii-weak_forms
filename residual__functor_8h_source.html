850  template <typename ResultScalarType>
851  return_type<ResultScalarType>
852  operator()(MeshWorker::ScratchData<dim, spacedim> &scratch_data,
853  const std::vector<SolutionExtractionData<dim, spacedim>>
854  &solution_extraction_data) const
855  {
856  // Follow the recipe described in the documentation:
857  // - Initialize helper.
858  // - Register independent variables and set the values for all fields.
859  // - Extract the sensitivities.
860  // - Use sensitivities in AD functor.
861  // - Register the definition of the residual field.
862  // - Compute residual field value, Jacobian, etc.
863  // - Later, extract the desired components of the residual field value,
864  // Jacobian etc.
865 
866  // Note: All user functions have the same parameterization, so we can
867  // use the same ADHelper for each of them. This does not restrict the
868  // user to use the same definition for the field itself at each
869  // QP!
870  ad_helper_type &ad_helper = get_mutable_ad_helper(scratch_data);
871  std::vector<Vector<scalar_type>> &values =
872  get_mutable_values(scratch_data, ad_helper);
873  std::vector<FullMatrix<scalar_type>> &Dvalues =
874  get_mutable_jacobians(scratch_data, ad_helper);
875 
876  const FEValuesBase<dim, spacedim> &fe_values =
877  scratch_data.get_current_fe_values();
878 
879  // In the HP case, we might traverse between cells with a different
880  // number of quadrature points. So we need to resize the output data
881  // accordingly.
882  if (values.size() != fe_values.n_quadrature_points ||
883  Dvalues.size() != fe_values.n_quadrature_points)
884  {
885  values.resize(fe_values.n_quadrature_points,
886  Vector<scalar_type>(
887  ad_helper.n_dependent_variables()));
888  Dvalues.resize(
889  fe_values.n_quadrature_points,
890  FullMatrix<scalar_type>(ad_helper.n_dependent_variables(),
891  ad_helper.n_independent_variables()));
892  }
893 
894  for (const auto q_point : fe_values.quadrature_point_indices())
895  {
896  ad_helper.reset();
897 
898  // Register the independent variables. The actual field solution at
899  // the quadrature point is fetched from the scratch_data cache. It
900  // is paired with its counterpart extractor, which should not have
901  // any indiced overlapping with the extractors for the other fields
902  // in the field_args.
903  OpHelper_t::ad_register_independent_variables(
904  ad_helper,
905  scratch_data,
906  solution_extraction_data,
907  q_point,
908  get_field_args(),
909  get_field_extractors());
910 
911  // Evaluate the functor to compute the residual field value.
912  // To do this, we extract all sensitivities and pass them directly
913  // in the user-provided function.
914  const residual_type residual_field_value =
915  OpHelper_t::ad_call_function(ad_helper,
916  function,

1257  template <typename ResultScalarType>
1258  return_type<ResultScalarType>
1259  operator()(MeshWorker::ScratchData<dim, spacedim> &scratch_data,
1260  const std::vector<SolutionExtractionData<dim, spacedim>>
1261  &solution_extraction_data) const
1262  {
1263  // Follow the recipe described in the documentation:
1264  // - Define some independent variables.
1265  // - Compute symbolic expressions that are dependent on the independent
1266  // variables.
1267  // - Create a optimizer to evaluate the dependent functions.
1268  // - Register symbols that represent independent variables.
1269  // - Register symbolic expressions that represent dependent functions.
1270  // - Optimize: Determine computationally efficient code path for
1271  // evaluation.
1272  // - Substitute: Pass the optimizer the numeric values that thee
1273  // independent variables to represent.
1274  // - Extract the numeric equivalent of the dependent functions from the
1275  // optimizer.
1276 
1277  // Note: All user functions have the same parameterization, so on the
1278  // face of it we can use the same BatchOptimizer for each of them. In
1279  // theory the user can encode the QPoint into the field function: this
1280  // current implementation restricts the user to use the same definition
1281  // for the field itself at each QP.
1282  const auto initialize_optimizer =
1283  [this](sd_helper_type<ResultScalarType> &batch_optimizer)
1284  {
1285  Assert(batch_optimizer.n_independent_variables() == 0,
1286  ExcMessage(
1287  "Expected the batch optimizer to be uninitialized."));
1288  Assert(batch_optimizer.n_dependent_variables() == 0,
1289  ExcMessage(
1290  "Expected the batch optimizer to be uninitialized."));
1291  Assert(batch_optimizer.values_substituted() == false,
1292  ExcMessage(
1293  "Expected the batch optimizer to be uninitialized."));
1294 
1295  // Create and register field variables (the independent variables).
1296  // We deal with the fields before the user data just in case
1297  // the users try to overwrite these field symbols. It shouldn't
1298  // happen, but this way its not possible to do overwrite what's
1299  // already in the map.
1300  Differentiation::SD::types::substitution_map symbol_map =
1301  OpHelper_t::template sd_get_symbol_map<sd_type>(
1302  get_symbolic_fields());
1303  if (user_symbol_registration_map)
1304  {
1305  Differentiation::SD::add_to_symbol_map(
1306  symbol_map,
1307  OpHelper_t::template sd_call_function<sd_type>(
1308  user_symbol_registration_map,
1309  get_symbolic_fields(),
1310  false /*compute_hash*/));
1311  }
1312  batch_optimizer.register_symbols(symbol_map);
1313 
1314  // The next typical few steps that precede function registration
1315  // have already been performed in the class constructor:
1316  // - Evaluate the functor to compute the total stored field.
1317  // - Compute the first derivatives of the field function.
1318  // - If there's some intermediate substitution to be done (modifying
1319  // the first derivatives), then do it before computing the second
1320  // derivatives.
1321  // (Why the intermediate substitution? If the first derivatives
1322  // represent the partial derivatives, then this substitution may be
1323  // done to ensure that the consistent linearization is given by the
1324  // second derivatives.)
1325  // - Differentiate the first derivatives (perhaps a modified form)
1326  // to get the second derivatives.
1327 
1328  // Register the dependent variables.
1329  OpHelper_t::template sd_register_functions<sd_type, residual_type>(
1330  batch_optimizer, residual);
1331  OpHelper_t::template sd_register_functions<sd_type, residual_type>(
1332  batch_optimizer, first_derivatives);
1333  };
1334 
1335  // Create and, if necessary, optimize a BatchOptimizer instance.
1336  // If the user has specified a cache, then this is only done once
1337  // per number of ScratchData times over the entire lifetime of the
1338  // cache.
1339  // If the user has not specified a cache, then this is done once
1340  // per number of ScratchData times for each assembly step (i.e.
1341  // with an additional multiplication factor like number of timesteps
1342  // times number of Newton iterations).
1343  sd_helper_type<ResultScalarType> &batch_optimizer =
1344  get_mutable_sd_batch_optimizer<ResultScalarType>(scratch_data);
1345  if (batch_optimizer.optimized() == false)
1346  {
1347  initialize_optimizer(batch_optimizer);
1348 
1349  // Finalize the optimizer.
1350  // If using the LLVM optimiser in a threaded environment, we have to
1351  // stagger initialisation so as not to cause some race condition
1352  // that leads to a segfault.
1353  if (optimization_method ==
1354  Differentiation::SD::OptimizerType::llvm &&
1355  MultithreadInfo::is_running_single_threaded() == false)
1356  {
1357  static std::mutex mutex;
1358  std::lock_guard<std::mutex> lock(mutex);
1359  batch_optimizer.optimize();
1360  }
1361  else
1362  {
1363  batch_optimizer.optimize();
1364  }
1365  }
1366 
1367  // Check that we've actually got a state that we can do some work with.
1368  Assert(batch_optimizer.n_independent_variables() > 0,
1369  ExcMessage("Expected the batch optimizer to be initialized."));
1370  Assert(batch_optimizer.n_dependent_variables() > 0,
1371  ExcMessage("Expected the batch optimizer to be initialized."));
1372 
1373  std::vector<std::vector<ResultScalarType>>
1374  &evaluated_dependent_functions =
1375  get_mutable_evaluated_dependent_functions<ResultScalarType>(
1376  scratch_data, batch_optimizer);
1377 
1378  const FEValuesBase<dim, spacedim> &fe_values =
1379  scratch_data.get_current_fe_values();
1380 
1381  // In the HP case, we might traverse between cells with a different
1382  // number of quadrature points. So we need to resize the output data
1383  // accordingly.
1384  if (evaluated_dependent_functions.size() !=
1385  fe_values.n_quadrature_points)
1386  {
1387  evaluated_dependent_functions.resize(
1388  fe_values.n_quadrature_points,
1389  std::vector<ResultScalarType>(
1390  batch_optimizer.n_dependent_variables()));
1391  }
1392 
1393  for (const auto q_point : fe_values.quadrature_point_indices())
1394  {
1395  // Substitute the field variables and whatever user symbols
1396  // are defined.
1397  // First we do the values from finite element fields,
1398  // followed by the values for user parameters, etc.
1399  Differentiation::SD::types::substitution_map substitution_map =
1400  OpHelper_t::template sd_get_substitution_map<sd_type,
1401  ResultScalarType>(
1402  scratch_data,
1403  solution_extraction_data,
1404  q_point,
1405  get_symbolic_fields(),
1406  get_field_args());
1407  if (user_substitution_map)
1408  {
1409  Differentiation::SD::add_to_substitution_map(
1410  substitution_map,
1411  user_substitution_map(scratch_data,
1412  solution_extraction_data,
1413  q_point));
1414  }
1415 
1416  // Perform the value substitution at this quadrature point
1417  batch_optimizer.substitute(substitution_map);
1418 
1419  // Extract evaluated data to be retrieved later.
1420  evaluated_dependent_functions[q_point] = batch_optimizer.evaluate();
1421  }
