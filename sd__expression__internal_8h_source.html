49  inline std::string
50  replace_protected_characters(const std::string &name)
51  {
52  // Allow SymEngine to parse this field as a string:
53  // Required for deserialization.
54  // It gets confused when there are numbers in the string name, and
55  // we have numbers and some protected characters in the expression
56  // name.
57  std::string out = name;
58  const auto replace_chars =
59  [&out](const char &old_char, const char &new_char)
60  { std::replace(out.begin(), out.end(), old_char, new_char); };
61  // replace_chars('0', 'A');
62  // replace_chars('1', 'B');
63  // replace_chars('2', 'C');
64  // replace_chars('3', 'D');
65  // replace_chars('4', 'E');
66  // replace_chars('5', 'F');
67  // replace_chars('6', 'G');
68  // replace_chars('7', 'H');
69  // replace_chars('8', 'I');
70  // replace_chars('9', 'J');
71  replace_chars(' ', '_');
72  replace_chars('(', '_');
73  replace_chars(')', '_');
74  replace_chars('{', '_');
75  replace_chars('}', '_');
76 
77  return out;
78  }
