60  // Check for duplicate field solution in the tuple
61  template <std::size_t J,
62  std::size_t I = 0,
63  typename FieldSolutionOp,
64  typename... SymbolicOpType>
65  static inline typename std::enable_if<
66  (I < J) &&
67  (FieldSolutionOp::rank ==
68  std::tuple_element_t<I, std::tuple<SymbolicOpType...>>::rank),
69  bool>::type
70  has_duplicate_previous_field_solution_op(
71  const FieldSolutionOp & field_solution,
72  const std::tuple<SymbolicOpType...> &symbolic_op_field_solutions)
73  {
74  // We cannot check the equivalence of entries like this:
75  // if (field_solution.as_expression() ==
76  // std::get<I>(symbolic_op_field_solutions).as_expression())
77  // {...}
78  // as the result of operator==() is actually a symbolic expression
79  // itself. In the case of equivalence, it can return something
80  // meaningful (as a boolean), but when the underlying expressions are
81  // different then it cannot be evaluated (as its actually trying to
82  // establish this equivalence of the returned values, not the symbolic
83  // expression itself). So, instead, we use substitution maps to decide
84  // if the two input expressions (or tensors of expressions) are
85  // identical, as merging a map of them should render a map of the size
86  // of one of the input maps.
87  //
88  // We also have to deal with the case where there are duplicate
89  // symbols. in the maps. This is not permitted by the
90  // make_symbol_map() function, but is by merge_substitution_maps().
91  const auto symbol_map_field = Differentiation::SD::make_symbol_map(
92  field_solution.as_expression());
93  const auto symbol_map_op = Differentiation::SD::make_symbol_map(
94  std::get<I>(symbolic_op_field_solutions).as_expression());
95  const auto symbol_map_merged =
96  Differentiation::SD::merge_substitution_maps(symbol_map_field,
97  symbol_map_op);
98  if (symbol_map_merged.size() == symbol_map_field.size())
99  {
100  return true;
101  }
102 
103  return has_duplicate_previous_field_solution_op<J, I + 1>(
104  field_solution, symbolic_op_field_solutions);
105  }

108  std::size_t I = 0,
109  typename FieldSolutionOp,
110  typename... SymbolicOpType>
111  static inline typename std::enable_if<
112  (I < J) &&
113  (FieldSolutionOp::rank !=
114  std::tuple_element_t<I, std::tuple<SymbolicOpType...>>::rank),
115  bool>::type
116  has_duplicate_previous_field_solution_op(
117  const FieldSolutionOp & field_solution,
118  const std::tuple<SymbolicOpType...> &symbolic_op_field_solutions)

322  std::size_t FieldIndex,
323  typename SymbolicOpField,
324  typename T = EnergyFunctor>

328  typename std::enable_if<is_ad_functor_op<T>::value>::type * =
329  nullptr) const

332  constexpr int spacedim = SymbolicOpField::space_dimension;
333 
334  using EnergyFunctorScalar_t = typename EnergyFunctor::scalar_type;
335  using FieldValue_t =
336  typename SymbolicOpField::template value_type<AssemblerScalar_t>;
337  using DiffOpResult_t = WeakForms::internal::Differentiation::
338  DiffOpResult<EnergyFunctorScalar_t, FieldValue_t>;
339 
340  using DiffOpValue_t = typename DiffOpResult_t::type;
341  using DiffOpFunction_t =
342  typename DiffOpResult_t::template function_type<dim>;

345  std::is_same<std::vector<DiffOpValue_t>,
346  typename DiffOpFunction_t::result_type>::value,
347  "Expected same result type.");
348 
349  // For AD types, the derivative_extractor will be a FEValues::Extractor.
350  const EnergyFunctor &functor = this->get_functor();
351  const auto & derivative_extractor =
352  functor.template get_derivative_extractor<FieldIndex>(
353  field); // Row extractor
354 
355  // The functor may only be temporary, so pass it in as a copy.
356  // The extractor is specific to this operation, so it definitely
357  // must be passed by copy.
358  return DiffOpResult_t::template get_functor<dim, spacedim>(
359  "Df",
360  "D(f)",
361  [functor, derivative_extractor](
362  MeshWorker::ScratchData<dim, spacedim> &scratch_data,
363  const std::vector<SolutionExtractionData<dim, spacedim>>
364  &solution_extraction_data)
365  {
366  (void)solution_extraction_data;
367  // We need to fetch the helper from Scratch (rather than passing
368  // it into this lambda function) to avoid working with the same copy
369  // of this object on multiple threads.
370  const auto &helper = functor.get_ad_helper(scratch_data);
371  // The return result from the differentiation is also not shared
372  // between threads. But we can reuse the same object many times
373  // since its stored in Scratch.
374  const std::vector<Vector<EnergyFunctorScalar_t>> &gradients =
375  functor.get_gradients(scratch_data);
376 
377  std::vector<DiffOpValue_t> out;
378  const FEValuesBase<dim, spacedim> &fe_values =
379  scratch_data.get_current_fe_values();
380  out.reserve(fe_values.n_quadrature_points);
381 
382  for (const auto q_point : fe_values.quadrature_point_indices())
383  out.emplace_back(
384  helper.extract_gradient_component(gradients[q_point],
385  derivative_extractor));
386 
387  return out;
388  },
389  functor,
390  field);

830  std::size_t I = 0,
831  std::size_t J = 0,
832  typename AssemblerType,
833  typename IntegralType,
834  typename... SymbolicOpType_1,
835  typename... SymbolicOpType_2>
836  inline typename std::enable_if < I < sizeof...(SymbolicOpType_1) &&
837  J<sizeof...(SymbolicOpType_2), void>::type
838  unpack_accumulate_bilinear_form_into(
839  AssemblerType & assembler,
840  const IntegralType & integral_operation,
841  const std::tuple<SymbolicOpType_1...> &symbolic_op_field_solutions_1,
842  const std::tuple<SymbolicOpType_2...> &symbolic_op_field_solutions_2)

845  using AssemblerScalar_t = typename AssemblerType::scalar_type;
846 
847  const auto &field_solution_test =
848  std::get<I>(symbolic_op_field_solutions_1);
849  const auto &field_solution_trial =
850  std::get<J>(symbolic_op_field_solutions_2);

852  bool perform_assembly = true;
853 
854  // We only allow one solution index, namely that pertaining to
855  // the current timestep / Newton iterate and active DoFHandler
856  // to be linearized.
857  if (field_solution_trial.solution_index !=
858  numbers::linearizable_solution_index)
859  {
860  perform_assembly = false;
861  }

863 # ifdef DEAL_II_WITH_SYMENGINE
864  // Support energies described by fully symbolic expressions:
865  // Need to check that we're not duplicating an operation that
866  // has already been performed due to the presence of an earlier,
867  // identical field-solution argument.
868  using FunctorType =
869  typename std::decay<decltype(this->get_functor())>::type;
870  if (is_sd_functor_op<FunctorType>::value &&
871  (internal::DuplicatePreviousFieldSolutionOpHelper::
872  has_duplicate_previous_field_solution_op<I>(
873  field_solution_test, symbolic_op_field_solutions_1) ||
874  internal::DuplicatePreviousFieldSolutionOpHelper::
875  has_duplicate_previous_field_solution_op<J>(
876  field_solution_trial, symbolic_op_field_solutions_2)))
877  {
878  perform_assembly = false;
879  }
880 # endif // DEAL_II_WITH_SYMENGINE
881 
882  if (perform_assembly)
883  {
884  const auto test_function =
885  WeakForms::internal::ConvertTo::test_function(
886  field_solution_test);
887  const auto trial_solution =
888  WeakForms::internal::ConvertTo::trial_solution(
889  field_solution_trial);
890 
891  // Since we derive from a potential, we can expect the contributions
892  // to the linear system to be symmetric.
893  const auto bilinear_form =
894  WeakForms::bilinear_form(
895  test_function,
896  get_functor_second_derivative<AssemblerScalar_t, I, J>(
897  field_solution_test, field_solution_trial),
898  trial_solution)
899  .symmetrize();

901  integral_operation.template integrate<AssemblerScalar_t>(
902  bilinear_form);
903 
904  if (OpSign == WeakForms::internal::AccumulationSign::plus)
905  {
906  assembler += integrated_bilinear_form;
907  }
908  else
909  {
910  Assert(OpSign == WeakForms::internal::AccumulationSign::minus,
911  ExcInternalError());
912  assembler -= integrated_bilinear_form;
913  }
914  }
915

917  // This effectively traverses the list of dependent fields, generating
918  // the bilinear forms associated with the linearization.
919  //
920  // Step 1: Linearize this linear form with respect to all field
921  // variables. This basically traverses the row subblock of the system
922  // and produces all column subblocks.
923  unpack_accumulate_bilinear_form_into<OpSign, I, J + 1>(
924  assembler,
925  integral_operation,
926  symbolic_op_field_solutions_1,
927  symbolic_op_field_solutions_2);
928  // Step 2: Only move on to the next row if we're at the zeroth column.
929  // This is because the above operation traverses all columns in a row.
930  if (J == 0)
931  unpack_accumulate_bilinear_form_into<OpSign, I + 1, J>(
932  assembler,
933  integral_operation,
934  symbolic_op_field_solutions_1,
935  symbolic_op_field_solutions_2);
936  }
