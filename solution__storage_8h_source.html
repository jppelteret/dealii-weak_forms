419  std::vector<SolutionExtractionData<dim, spacedim>>
420  get_solution_extraction_data(
421  MeshWorker::ScratchData<dim, spacedim> &scratch_data,
422  const DoFHandlerType2<dim, spacedim> & dof_handler) const
423  {
424  std::vector<SolutionExtractionData<dim, spacedim>>
425  solution_extraction_data;
426  solution_extraction_data.reserve(n_solution_vectors());
427 
428  for (unsigned int t = 0; t < n_solution_vectors(); ++t)
429  {
430  if (uses_same_dof_handler(t, dof_handler))
431  {
432  constexpr const bool uses_external_dofhandler = false;
433  solution_extraction_data.emplace_back(scratch_data,
434  get_solution_name(t),

436  }
437  else
438  {
439  Assert(
440  (!std::is_same<DoFHandlerType, std::nullptr_t>::value),
441  ExcMessage(

443  "is not associated with another DoFHandler. You need to use the "
444  "constructor that takes in a DoFHandler in order to use this "
445  "functionality."));
446 
447  GeneralDataStorage &cache =
448  scratch_data.get_general_data_storage();
449  MeshWorker::ScratchData<dim, spacedim> &external_scratch_data =
450  cache.template get_object_with_name<
451  MeshWorker::ScratchData<dim, spacedim>>(

453
