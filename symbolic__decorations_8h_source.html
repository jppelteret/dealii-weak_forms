561  template <typename UnaryFunctorOp, typename... UnaryFieldOps>
562  std::string
563  symbolic_op_derivative_as_ascii(
564  const UnaryFunctorOp & unary_functor,
565  const std::tuple<UnaryFieldOps...> &symbolic_op_field_solutions) const
566  {
567  const SymbolicDecorations &decorator = *this;
568  constexpr int n_diff_operations =
569  std::tuple_size<std::tuple<UnaryFieldOps...>>::value;
570 
571  if (n_diff_operations == 0)
572  return unary_functor.as_ascii(decorator);
573 
574  // Form the numerator of the differential notation
575  std::string out = "d";
576  if (n_diff_operations > 1)
577  {
578  out += dealii::Utilities::to_string(n_diff_operations);
579  }
580  out += "(" + unary_functor.as_ascii(decorator) + ")";
581 
582  // Form the denominator of the differential notation
583  out += "/";
584  if (n_diff_operations > 1)
585  {
586  out += "(";
587  }
588  out += differential_expansion_of_unary_field_ops_as_ascii(
589  symbolic_op_field_solutions);
590  if (n_diff_operations > 1)
591  {
592  out += ")";
593  }
594 
595  return out;
596  }

720  void
721  deduplicate_vector_of_strings(std::vector<std::string> &str_vec) const
722  {
723  std::unordered_set<std::string> already_visited;
724 
725  auto end = std::remove_if(str_vec.begin(),
726  str_vec.end(),
727  [&already_visited](const std::string &i)
728  { return !already_visited.insert(i).second; });
729 
730  str_vec.erase(end, str_vec.end());
731  }
732 
733  std::string
734  concatenate_vector_of_strings(const std::vector<std::string> &str_vec,
735  const std::string &delimiter = ", ") const
736  {
737  std::stringstream stream;
738  std::copy(str_vec.cbegin(),
739  str_vec.cend(),
740  std::ostream_iterator<std::string>(stream, delimiter.c_str()));
741 
742  // Remove trailing delimiter
743  std::string out = stream.str();
744  out.resize(out.size() - delimiter.size());
745  return out;
746  }
