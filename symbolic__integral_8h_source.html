53  template <bool isCellPredicate>

56  template <int dim, int spacedim>
57  using tria_iterator = typename std::conditional<
58  isCellPredicate,
59  typename Triangulation<dim, spacedim>::cell_iterator,
60  typename Triangulation<dim, spacedim>::face_iterator>::type;
61 
62  template <int dim, int spacedim>
63  using dofhandler_iterator = typename std::conditional<
64  isCellPredicate,
65  typename DoFHandler<dim, spacedim>::cell_iterator,
66  typename DoFHandler<dim, spacedim>::face_iterator>::type;
67 
68  template <int dim, int spacedim>
69  using TriangulationPredicateFunctionType =
70  std::function<bool(const tria_iterator<dim, spacedim> &)>;
71 
72  template <int dim, int spacedim>
73  using DoFHandlerPredicateFunctionType =
74  std::function<bool(const dofhandler_iterator<dim, spacedim> &)>;
75 
76  template <class PredicateType>
77  PredicateKeeper(const PredicateType &predicate)
78  : tria_predicate_11([predicate](const tria_iterator<1, 1> &it)
79  { return predicate(it); })

81  { return predicate(it); })

83  { return predicate(it); })

85  { return predicate(it); })

87  { return predicate(it); })

89  { return predicate(it); })
90  , dofhandler_predicate_11(

92  { return predicate(it); })
